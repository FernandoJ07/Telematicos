# Configuración del kernel para el servidor esclavo
# Salida del comando: sudo sysctl -p /etc/ufw/sysctl.conf

# Habilitar reenvío de paquetes IPv4
net.ipv4.ip_forward = 1

# Deshabilitar redirecciones ICMP (seguridad)
net.ipv4.conf.all.accept_redirects = 0
net.ipv4.conf.default.accept_redirects = 0
net.ipv6.conf.all.accept_redirects = 0
net.ipv6.conf.default.accept_redirects = 0

# Ignorar broadcasts de ping (seguridad)
net.ipv4.icmp_echo_ignore_broadcasts = 1

# Ignorar respuestas ICMP erróneas (seguridad)
net.ipv4.icmp_ignore_bogus_error_responses = 1

# No ignorar todos los pings (permitir ping normal)
net.ipv4.icmp_echo_ignore_all = 0

# Deshabilitar logging de paquetes marcianos (para producción)
net.ipv4.conf.all.log_martians = 0
net.ipv4.conf.default.log_martians = 0
